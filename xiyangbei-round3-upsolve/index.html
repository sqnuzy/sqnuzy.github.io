【小羊肖恩】小羊杯 Round 3_ACM/NOI/CSP/CCPC/ICPC算法编程高难度练习赛_牛客竞赛OJ

M:

题意

现有一个字符串集合。需要支持以下的操作

1.对集合中所有的字符串，新增后缀字符。

2.插入新字符串。

3.处理查询：对于字符串 s ，s 和字符串集合中所有的字符串的最长公共后缀的长度之和是多少？最长公共后缀指的是同时是两个字符串的后缀的最长的字符串。

我的问题

不会怎么在字典树中加前缀

正解

换一下根即可

易错点

在修改字典树的前缀题中，根需要从1开始，否则会有误判的错误，同时每次查询的时候要从当前根开始

代码

```cpp #include <bits/stdc++.h> using namespace std;

using ll = long long;

const int N = 1500000 + 10; const int SIGMA = 26;

int to[N][SIGMA], cnt[N], idx; int root, tot;

int id(char c) { return c - ‘a’; }

void init() { for (int i = 0; i <= idx; i++) { cnt[i] = 0; for (int j = 0; j < SIGMA; j++) to[i][j] = 0; } idx = 1; root = 1; tot = 0; }

void add(char ch) { if (tot == 0) return; int u = id(ch); ++idx; to[idx][u] = root; cnt[idx] = cnt[root]; root = idx; }

void insert(const string &s) { int p = root; tot++; cnt[p]++; for (char c : s) { int u = id(c); if (!to[p][u]) to[p][u] = ++idx; p = to[p][u]; cnt[p]++; } }

ll query(const string &s) { int p = root; ll ans = 0; for (char c : s) { int u = id(c); if (!to[p][u]) break; p = to[p][u]; ans += cnt[p]; } return ans; }

void solve() { init();

int q;
cin >> q;
while (q--) {
    int op;
    string s;
    cin >> op >> s;
    if (op == 1) {
        add(s[0]);
    } else if (op == 2) {
        reverse(s.begin(), s.end());
        insert(s);
    } else {
        reverse(s.begin(), s.end());
        cout << query(s) << "\n";
    }
}

}

int main() { ios::sync_with_stdio(false); cin.tie(0); init(); int _T = 1; // cin >> _T; while (_T–) solve(); return 0; }
